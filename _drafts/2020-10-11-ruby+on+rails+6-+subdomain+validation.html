validates :subdomain, 

exclusion: {in: RESERVED_SUBDOMAINS, message: "%{value} is reserved."}, 

format: {with: /\A[a-zA-Z0-9]+[a-zA-Z0-9\-_]*[a-zA-Z0-9]+\Z/, 

message: "must be at least 2 characters and alphanumeric", 

allow_blank: true}


RESERVED_SUBDOMAINS = %w[app help support]
